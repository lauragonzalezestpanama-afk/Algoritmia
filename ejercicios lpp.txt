Real base, altura, area, perimetro

Inicio
  Escriba "Ingrese la base del rectángulo: "
  Lea base

  Escriba "Ingrese la altura del rectángulo: "
  Lea altura

  area <- base * altura
  perimetro <- 2 * (base + altura)

  Escriba "Área: ", area
  Escriba "Perímetro: ", perimetro
Fin
